-- TABLA CLIENTE_DG
CREATE TABLE CLIENTE_DG (
    idC NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tipoDocumC VARCHAR2(20) NOT NULL,
    nroDocumC VARCHAR2(20) NOT NULL,
    nombresC VARCHAR2(100) NOT NULL,
    paternoC VARCHAR2(100) NOT NULL,
    maternoC VARCHAR2(100) NOT NULL,
    celularC VARCHAR2(9) NOT NULL
);

-- TABLA CATEGORIA_DG
CREATE TABLE CATEGORIA_DG (
    idCat NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombreCat VARCHAR2(100) NOT NULL,
    descripcionCat VARCHAR2(200)
);

-- TABLA PRODUCTO_DG
CREATE TABLE PRODUCTO_DG (
    idP NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombreP VARCHAR2(100) NOT NULL,
    descripcionP VARCHAR2(200),
    precioReferencialP NUMBER(10,2) NOT NULL,
    idCat NUMBER NOT NULL,
    CONSTRAINT FK_PRODUCTO_DG_CATEGORIA
    FOREIGN KEY (idCat) REFERENCES CATEGORIA_DG(idCat)
);

-- TABLA COMPROBANTE_DG
CREATE TABLE COMPROBANTE_DG (
    idComp NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fechaHoraComp TIMESTAMP NOT NULL,
    totalComp NUMBER(10,2) NOT NULL,
    idC NUMBER NOT NULL,
    CONSTRAINT FK_COMPROBANTE_DG_CLIENTE
    FOREIGN KEY (idC) REFERENCES CLIENTE_DG(idC)
);

-- TABLA DETALLE_DG
CREATE TABLE DETALLE_DG (
    idDet NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    cantidadDet NUMBER NOT NULL,
    precioRealUnitarioDet NUMBER(10,2) NOT NULL,
    subtotalDet NUMBER(10,2) NOT NULL,
    idComp NUMBER NOT NULL,
    idP NUMBER NOT NULL,
    CONSTRAINT FK_DETALLE_DG_COMPROBANTE
    FOREIGN KEY (idComp) REFERENCES COMPROBANTE_DG(idComp),
    CONSTRAINT FK_DETALLE_DG_PRODUCTO
    FOREIGN KEY (idP) REFERENCES PRODUCTO_DG(idP)
);
