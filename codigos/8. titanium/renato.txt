-- 1. Limpieza de tablas previas en ejercicio8
SET FOREIGN_KEY_CHECKS = 0;
DROP TABLE IF EXISTS Asistencia;
DROP TABLE IF EXISTS GrupoDisciplina;
DROP TABLE IF EXISTS Disciplina;
DROP TABLE IF EXISTS Cliente;
SET FOREIGN_KEY_CHECKS = 1;

-- 2. Creación de tablas (Cambiando BIT por TINYINT y IDENTITY por AUTO_INCREMENT)
CREATE TABLE Cliente (
    idC INT AUTO_INCREMENT PRIMARY KEY,
    tipoDocumC VARCHAR(20) NOT NULL,
    nroDocumC VARCHAR(20) NOT NULL,
    nombresC VARCHAR(100) NOT NULL,
    paternoC VARCHAR(100) NOT NULL,
    maternoC VARCHAR(100) NOT NULL,
    generoC VARCHAR(10) NOT NULL,
    fechaNacimC DATE NOT NULL,
    enfremedadesC VARCHAR(250) NULL,
    celularC VARCHAR(9) NOT NULL
) ENGINE=InnoDB;

CREATE TABLE Disciplina (
    idD INT AUTO_INCREMENT PRIMARY KEY,
    nombreD VARCHAR(100) NOT NULL,
    descripcionD VARCHAR(250) NULL
) ENGINE=InnoDB;

CREATE TABLE GrupoDisciplina (
    idG INT AUTO_INCREMENT PRIMARY KEY,
    nombreG VARCHAR(100) NOT NULL,
    fechainicioG DATE NOT NULL,
    fechaFinG DATE NOT NULL,
    idDisciplina INT NOT NULL,
    CONSTRAINT FK_Grupo_Disciplina FOREIGN KEY (idDisciplina) REFERENCES Disciplina(idD)
) ENGINE=InnoDB;

CREATE TABLE Asistencia (
    idA INT AUTO_INCREMENT PRIMARY KEY,
    fechaA DATE NOT NULL,
    diaA VARCHAR(15) NOT NULL,
    esferiadoA TINYINT(1) NOT NULL, -- MySQL usa TINYINT en lugar de BIT
    asistioA TINYINT(1) NOT NULL,
    horaEntradaA TIME NULL,
    idCliente INT NOT NULL,
    idGrupo INT NOT NULL,
    CONSTRAINT FK_Asist_Cliente FOREIGN KEY (idCliente) REFERENCES Cliente(idC),
    CONSTRAINT FK_Asist_Grupo FOREIGN KEY (idGrupo) REFERENCES GrupoDisciplina(idG)
) ENGINE=InnoDB;
