-- TABLA CLIENTEENV
CREATE TABLE ClienteEnv (
    IdCliente NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    TipoDocum VARCHAR2(20) NOT NULL,
    NroDocum VARCHAR2(20) NOT NULL,
    Nombres VARCHAR2(100) NOT NULL,
    Paterno VARCHAR2(100) NOT NULL,
    Materno VARCHAR2(100) NOT NULL,
    Celular VARCHAR2(9) NOT NULL,
    IdReferencia NUMBER,
    CONSTRAINT FK_ClienteEnv_Referido
    FOREIGN KEY (IdReferencia) REFERENCES ClienteEnv(IdCliente)
);

-- TABLA COMPROBANTEENV
CREATE TABLE ComprobanteEnv (
    idComp NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fechaHoraComp TIMESTAMP NOT NULL,
    totalComp NUMBER(10,2) NOT NULL,
    OrigenComp VARCHAR2(100) NOT NULL,
    DestinoComp VARCHAR2(100) NOT NULL,
    IdCliente NUMBER NOT NULL,
    CONSTRAINT FK_ComprobanteEnv_Cliente
    FOREIGN KEY (IdCliente) REFERENCES ClienteEnv(IdCliente)
);

-- TABLA PAQUETEENV
CREATE TABLE PaqueteEnv (
    idP NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    descripcionP VARCHAR2(200) NOT NULL,
    pesoP NUMBER(10,2) NOT NULL
);

-- TABLA DETALLEENV
CREATE TABLE DetalleEnv (
    idDet NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    precioPorkiloDet NUMBER(10,2) NOT NULL,
    subtotalDet NUMBER(10,2) NOT NULL,
    idComp NUMBER NOT NULL,
    idP NUMBER,
    CONSTRAINT FK_DetalleEnv_Comprobante
        FOREIGN KEY (idComp) REFERENCES ComprobanteEnv(idComp),
    CONSTRAINT FK_DetalleEnv_Paquete
        FOREIGN KEY (idP) REFERENCES PaqueteEnv(idP)
);
