-- 1. Asegurar limpieza (ejecuta esto dentro de tu base de datos)
SET FOREIGN_KEY_CHECKS = 0;
DROP TABLE IF EXISTS T_Asistencia;
DROP TABLE IF EXISTS T_Empleado;
DROP TABLE IF EXISTS T_Turno;
DROP TABLE IF EXISTS T_Anio;
SET FOREIGN_KEY_CHECKS = 1;

-- 2. Creación de tablas (Nombres descriptivos para evitar errores de ruta)
CREATE TABLE T_Anio (
    id_anio INT AUTO_INCREMENT PRIMARY KEY,
    inicio_anio DATE NOT NULL,
    fin_anio DATE NOT NULL
) ENGINE=InnoDB;

CREATE TABLE T_Turno (
    id_turno INT AUTO_INCREMENT PRIMARY KEY,
    nombre_turno VARCHAR(50) NOT NULL,
    inicio_turno TIME NOT NULL,
    fin_turno TIME NOT NULL,
    id_anio INT NOT NULL,
    CONSTRAINT FK_Turno_Anio FOREIGN KEY (id_anio) REFERENCES T_Anio(id_anio)
) ENGINE=InnoDB;

CREATE TABLE T_Empleado (
    id_empleado INT AUTO_INCREMENT PRIMARY KEY,
    tipo_doc VARCHAR(20) NOT NULL,
    nro_doc VARCHAR(20) NOT NULL,
    nombres VARCHAR(100) NOT NULL,
    paterno VARCHAR(100) NOT NULL,
    materno VARCHAR(100) NOT NULL,
    celular VARCHAR(9) NOT NULL,
    id_supervisor INT NULL,
    CONSTRAINT FK_Empleado_Supervisor FOREIGN KEY (id_supervisor) REFERENCES T_Empleado(id_empleado)
) ENGINE=InnoDB;

CREATE TABLE T_Asistencia (
    id_asistencia INT AUTO_INCREMENT PRIMARY KEY,
    fecha_asist DATE NOT NULL,
    dia_asist VARCHAR(15) NOT NULL,
    es_feriado BOOLEAN NOT NULL,
    asistio BOOLEAN NOT NULL,
    tardo BOOLEAN NOT NULL,
    hora_entrada TIME NULL,
    hora_salida TIME NULL,
    id_empleado INT NOT NULL,
    id_turno INT NOT NULL,
    CONSTRAINT FK_Asistencia_Empleado FOREIGN KEY (id_empleado) REFERENCES T_Empleado(id_empleado),
    CONSTRAINT FK_Asistencia_Turno FOREIGN KEY (id_turno) REFERENCES T_Turno(id_turno)
) ENGINE=InnoDB;
