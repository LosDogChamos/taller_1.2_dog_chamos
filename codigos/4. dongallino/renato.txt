-- 1. Limpieza de tablas previas en ejercicio4
SET FOREIGN_KEY_CHECKS = 0;
DROP TABLE IF EXISTS Detalle;
DROP TABLE IF EXISTS Comprobante;
DROP TABLE IF EXISTS Producto;
DROP TABLE IF EXISTS Categoria;
DROP TABLE IF EXISTS Cliente;
SET FOREIGN_KEY_CHECKS = 1;

-- 2. Creación de tablas (sin crear base de datos nueva)
CREATE TABLE Cliente (
    idC INT AUTO_INCREMENT PRIMARY KEY,
    tipoDocumC VARCHAR(20) NOT NULL,
    nroDocumC VARCHAR(20) NOT NULL,
    nombresC VARCHAR(100) NOT NULL,
    paternoC VARCHAR(100) NOT NULL,
    maternoC VARCHAR(100) NOT NULL,
    celularC VARCHAR(9) NOT NULL
) ENGINE=InnoDB;

CREATE TABLE Categoria (
    idCat INT AUTO_INCREMENT PRIMARY KEY,
    nombreCat VARCHAR(100) NOT NULL,
    descripcionCat VARCHAR(200) NULL
) ENGINE=InnoDB;

CREATE TABLE Producto (
    idP INT AUTO_INCREMENT PRIMARY KEY,
    nombreP VARCHAR(100) NOT NULL,
    descripcionP VARCHAR(200) NULL,
    precioReferencialP DECIMAL(10, 2) NOT NULL,
    idCat INT NOT NULL,
    CONSTRAINT FK_Producto_Categoria FOREIGN KEY (idCat) REFERENCES Categoria(idCat)
) ENGINE=InnoDB;

CREATE TABLE Comprobante (
    idComp INT AUTO_INCREMENT PRIMARY KEY,
    fechaHoraComp DATETIME NOT NULL,
    totalComp DECIMAL(10, 2) NOT NULL,
    idC INT NOT NULL,
    CONSTRAINT FK_Comprobante_Cliente FOREIGN KEY (idC) REFERENCES Cliente(idC)
) ENGINE=InnoDB;

CREATE TABLE Detalle (
    idDet INT AUTO_INCREMENT PRIMARY KEY,
    cantidadDet INT NOT NULL,
    precioRealUnitarioDet DECIMAL(10, 2) NOT NULL,
    subtotalDet DECIMAL(10, 2) NOT NULL,
    idComp INT NOT NULL,
    idP INT NOT NULL,
    CONSTRAINT FK_Detalle_Comprobante FOREIGN KEY (idComp) REFERENCES Comprobante(idComp),
    CONSTRAINT FK_Detalle_Producto FOREIGN KEY (idP) REFERENCES Producto(idP)
) ENGINE=InnoDB;
